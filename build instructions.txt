11. qmake "INCLUDEPATH+=C:\Entwicklung\MySQL\Server5.1\include" "LIBS+=C:\Entwicklung\MySQL\Server5.1\lib\opt\libmysql.a" mysql.pro
12. cd %QTDIR%
13. mingw32-make
